 <span data-ttu-id="a829b-107">Aşağıdaki örnek CS0206 oluşturur:</span><span class="sxs-lookup"><span data-stu-id="a829b-107">The following sample generates CS0206:</span></span>  
  
```csharp  
// CS0206.cs  
public class MyClass  
{  
    public static int P  
    {  
        get  
        {  
            return 0;  
        }  
        set  
        {  
        }  
    }  
  
    public static void MyMeth(ref int i)  
    // public static void MyMeth(int i)

    }

    {  
        MyMeth(ref P);   // CS0206  
        // try the following line instead  
        // MyMeth(P);   // CS0206  
    }  
}  
```
